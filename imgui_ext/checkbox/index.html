Example

#[derive(imgui_ext::Gui)]
struct Checkboxes {
    // All parameters are optional.
    #[imgui(checkbox)]
    turbo: bool,

    // Optionally, you can override the label:
    #[imgui(checkbox(label = "Checkbox!"))]
    check: bool,
}
